#include<iostream>
using namespace std;
class complex
{
    int a,b;
    public:
    complex(int,int);
    void print()
    {
        cout<<"No is:"<<a<<"+"<<b<<"i"<<endl;
    }
    
};
complex::complex (int x,int y)
{
    a=x;
    b=y;
    
}
int main()
{
    complex d(4,5);//implict call
    complex e=complex(8,9);//explicit call
    d.print();
    e.print();
}
